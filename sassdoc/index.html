    flexposition

    @mixin flexposition($direction, $justify, $align, $wrap, $gap) { ... }

    Description

    mixin para posicionar elementos con flexbox

    Parameters

    parameter Nameparameter Descriptionparameter Typeparameter Default value
    $direction

    direccion del contenedor flex

    		String— none
    $justify

    alineacion horizontal de los elementos

    		String— none
    $align

    alineacion vertical de los elementos

    		String— none
    $wrap

    si los elementos se ajustan al contenedor

    		String— none
    $gap

    espacio entre los elementos

    		Integer— none

    Example

    .menu {
    @include flexposition(row, space-between, center);

    &__options {
        @include flexposition(row, space-between, center);
    }
}

    @mixin gridtemplate($type, $min, $max, $row-gap, $column-gap) { ... }

    Description

    Parameters

    parameter Nameparameter Descriptionparameter Typeparameter Default value
    $type

    tipo de grid

    		String— none
    $min

    tamaño minimo de los elementos

    		String— none
    $max

    tamaño maximo de los elementos

    		String— none
    $row-gap

    espacio entre las filas

    		String— none
    $column-gap

    espacio entre las columnas

    		Integer— none

    Example

    @include gridtemplate(auto-fit, 250px, 1fr, 1.5rem, 1rem);
}
